function fetch() {

	var xhr, name, quantity, description, category, price, target, file; 

	xhr = new XMLHttpRequest();

	target = document.getElementById("status");

	name = document.getElementById("name").value;
    quantity = document.getElementById("quantity").value;
    description = document.getElementById("description").value;
    category = document.getElementById("category").value;
    price = document.getElementById("price").value;

    var vars = "name="+name+"&quantity="+quantity+"&description="+description+"&category="+category+"&price="+price;


    changeListener = function () {
		if(xhr.readyState == 4 && xhr.status == 200) {
		   target.innerHTML = xhr.responseText;
		} else {
			target.innerHTML = "<p>Something went wrong.</p>";
		}
	};

	xhr.open("POST", "ajax/sql/addProductSQL.php", true);
	xhr.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
	xhr.onreadystatechange = changeListener;
	xhr.send(vars);


}
